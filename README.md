#include <iostream>

using namespace std;

int main()
{

    cout << "Masukkan panjang persegi panjang :";

    cout<< "Masukkan lebar persegi panjang :";
    cin>>l;

    luas=p*l;
    cout<< "Luas persegi panjang adalah :"<<luas;
    return 0;
}
